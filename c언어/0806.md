# 사칙연산

```c
#include <stdio.h>

int main() {
    int i, j;
    printf("정수 2개 입력 :");
    scanf("%d%d", &i, &j);
    printf("%d + %d = %d\n", i, j, (i + j));
    printf("%d - %d = %d\n", i, j, (i - j));
    printf("%d * %d = %d\n", i, j, (i * j));
    printf("%d / %d = %.2f\n", i, j, ((float)i / j));
    printf("%d %% %d = %d\n", i, j, (i % j));
    return 0;
}
```
-결과
```
정수 2개 입력 :8 + 4 = 12
8 - 4 = 4
8 * 4 = 32
8 / 4 = 2.00
8 % 4 = 0
```

```c
#include <stdio.h>
int main() {
    int i, j;
    printf("정수 2개 입력 :");
    scanf("%d%d", &i, &j);
    printf("%d > %d = %d\n", i, j, (i > j));
    printf("%d >= %d = %d\n", i, j, (i >= j));
    printf("%d < %d = %d\n", i, j, (i < j));
    printf("%d <= %d = %d\n", i, j, (i <= j));
    printf("%d == %d = %d\n", i, j, (i == j));
    printf("%d != %d = %d\n", i, j, (i != j));
    return 0;
}
```

