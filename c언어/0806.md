# 사칙연산

```c
#include <stdio.h>

int main() {
    int i, j;
    printf("정수 2개 입력 :");
    scanf("%d%d", &i, &j);
    printf("%d + %d = %d\n", i, j, (i + j));
    printf("%d - %d = %d\n", i, j, (i - j));
    printf("%d * %d = %d\n", i, j, (i * j));
    printf("%d / %d = %.2f\n", i, j, ((float)i / j));
    printf("%d %% %d = %d\n", i, j, (i % j));
    return 0;
}
```
-결과
```
정수 2개 입력 :8 + 4 = 12
8 - 4 = 4
8 * 4 = 32
8 / 4 = 2.00
8 % 4 = 0
```

# 논리연산
```c
#include <stdio.h>
int main() {
    int i, j;
    printf("논리값 2개 입력 :");
    scanf("%d%d", &i, &j);
    printf("\n!(%d) = %d\n", i, !i);
    printf("%d && %d = %d\n", i, j, (i && j));
    printf("%d || %d = %d\n", i, j, (i || j));
 
    return 0;
}
```
```
-결과 
논리값 2개 입력 :
!(0) = 1
0 && 1 = 0
0 || 1 = 1
```

- 연산잔
```c
#include <stdio.h>
int main() {
    int i = 0, j;
    i = i + 1;
    printf("i = %d\n", i);
    i = 0;
    i += 1;
    printf("i = %d\n", i);
    i = 0;
    ++i;
    printf("i = %d\n", i);
    i = 0;
    i++;
    printf("i = %d , j = %d\n", i, j);
    i = 0;
    j = ++i;
    printf("i = %d , j = %d\n", i, j);
    
    return 0;
}
```
-결과 
```
i = 1
i = 1
i = 1
i = 1 , j = 0
i = 1 , j = 1
```
