```c
#include <stdio.h>
int main() {
int number1, number2;
printf("첫번째 변수 : ");
 scanf("%d" , &number1);
printf("두번째 변수 :");
scanf("%d" , &number2);
printf("%d + %d = %d\n", number1 , number2, (number1 + number2));
printf("%d - %d = %d\n", number1 , number2, (number1 - number2));
printf("%d * %d = %d\n", number1 , number2, (number1 * number2));
printf("%d / %d = %.2f\n", number1 , number2, ((float)number1 / number2));
printf("%d %% %d = %d\n", number1 , number2, (number1 % number2));
    return 0;
}
```

- 윤년 (leap year)
```python
#윤년은 4의 배구이고 100의 배수는  아니고 400의 배수이면 윤년입니다.
#년도를 입력받아 윤년인지 평년(common year)인지를 풀력하는
#프로그램을 작성하세요!

year = int(input("년도 입력 : "))

if year%4==0 and year%100!=0 or year%400==0:
        print("윤년(leep year)입니다")
else:
        print("평년(common year)입니다")
```
- `==`는 수학적을 같다라는 으미 `!=`는 같지않다라는 의미
- `=` 변수를 대입하는 것

# 반복 곱하기 
```python
n = int(input("양의 정수 입력 : "))
fact = 1

for i in range(1, n+1):
        fact *= i

print("{}! = {}".format(n, fact))
```
- 결과
양의 정수 입력 : 5
5! = 120

- 보이게
```python
n = int(input("양의 정수 입력 : "))
fact = 1

for i in range(n , 0, -1):
        if i !=1:
                print("{} X".format(i), end="")
        else:
                print("{} = ".format(i), end="")
        fact *= i

print(fact)
```
